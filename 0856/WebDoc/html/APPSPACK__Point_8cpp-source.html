00128 bool APPSPACK::Point::operator<(const Point& other) const
00129 {
00130   // Case I: Equal tags
00131   if (tag == other.tag)
00132     return false;
00133 
00134   // Case II: This point does not have a valid function value
00135   if ((state != Cached) && (state != Evaluated))
00136     return false;
00137 
00138   // Case III: The other point does not have a valid function value
00139   if ((other.state != Cached) && (other.state != Evaluated))
00140     return true;
00141 
00142   // Case IV: f(x) < f(y)
00143   if (f < other.f)              
00144     return true;
00145 
00146   // Case III: f(x) > f(y)
00147   if (f > other.f)
00148     return false;
00149 
00150   // Case IV: f(x) = f(y)
00151   return (tag < other.tag);
00152 }
